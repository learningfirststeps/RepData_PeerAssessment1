### "Create a new dataset that is equal to the original dataset but with the missing data filled in." *quote from assignment." I use the rounded mean for that 5-minute interval across all dataset to fill NAs.
restoreddata <- rawdata
for (k in 1:nrow(restoreddata)) {
        if ( is.na( restoreddata[k,"steps"] ) ) {
                restoreddata[k,"steps"] <- p[ as.character( restoreddata[k,"interval"] ) ]
        }
}
